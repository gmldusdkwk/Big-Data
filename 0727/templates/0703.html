def Avoiding_Duplicate():
    b = list()
    while(True):
        a = input()
        if a == ' ':
            break
        b.append(a)
        #print(b)
        for i in range(len(b)-1):
            if len(b) == 1:
                break
            if b[i] == a:
                #print(True)
                b.pop(len(b)-1)
    print(b)

    d = list()
