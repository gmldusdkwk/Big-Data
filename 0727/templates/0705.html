def Compute_Grade():
    result = 0
    i = 0
    while(True):
        a = input()
        if(a == ' '):
            break
        i += 1
        if(a == 'A+' or a == 'A'):
            result += 4.0
        elif(a == 'A-'):
            result += 3.7
        elif(a == 'B+'):
            result += 3.3
        elif(a == 'B'):
            result += 3.0
        elif(a == 'B-'):
            result += 2.7
        elif(a == 'C+'):
            result += 2.3
        elif(a == 'C'):
            result += 2.0
        elif(a == 'C-'):
            result += 1.7
        elif(a == 'D+'):
            result += 1.3
        elif(a == 'D'):
            result += 1.
        elif(a == 'F'):
            result += 0
        else:
            continue
    print(result/i)

Exercise 72: Is a String a Palindrome?¶

  1. mid 값을 정한다.
    • 글자수가 짝수 -> 4/2 = 2 그냥 그대로 하면 된다.
    • 홀수 -> 몫만 챙겨야 한다.
  2. 문자열의 앞부터 그리고 문자열의 뒤부터 확인해야 한다.
    • 짝수: 미드 값 포함
    • 홀수: 미드 값 제외
In [41]:
def Is_a_String_Palidrome(a):
    #print(len(a))
    if len(a) % 2 == 0:
        mid = len(a)//2
        #print(mid)
        for i in range(mid):
            if a[i] != a[-(i+1)]:
                return False
    else:
        mid = len(a)//2
        for i in range(mid):
            if a[i] != a[-(i+1)]:
                return False
    return True

def B_to_D(a):
    ans = 0;
    for i in range(len(a)):
        mul = 1;
        for j in range(len(a) - i - 1):
            mul *= 2
        ans += int(a[i]) * mul
    print(ans)
